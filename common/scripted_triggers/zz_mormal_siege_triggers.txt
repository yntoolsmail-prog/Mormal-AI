#  mormal_normal_maa_regiments_at_least_trigger ( COUNT = N ):
#    «у персонажа >= N полков ОБЫЧНЫХ (не осадных) латников».

#    Перечислены все НЕосадные базовые типы из файлов 1.19
#    (вкл. peasant_militia и nomadic_horde — для полноты; орде наши 4 типа
#    всё равно запрещены ванильным can_recruit). Потолок 12 полков на ОДИН
#    базовый тип практически недостижим (лимит полков ~ярус+бонусы).
#
#  mormal_ai_siege_recruit_gate_trigger:
#    разрешение ИИ нанять ЕЩЁ ОДИН осадный полк по формуле спроса движка
#    (vanilla/00_ai.txt:639-653: SUBTRACT_NORMAL_SUB_REGIMENTS_FOR_SIEGE_PURPOSES=3,
#     NORMAL_SUB_REGIMENTS_PER_SIEGE_SUB_REGIMENT=5):
#      k-й осадный полк разрешён при обычных >= 3 + 5*(k-1):
#      1-й — c 3 обычных, 2-й — с 8, 3-й — с 13, 4-й — с 18; 5-й+ запрещён
#      (недостижимо по лимиту полков). Счёт осадных — по базовому типу
#      siege_weapon, т.е. ловит и унаследованные/чужие осадные полки.
# =============================================================================

mormal_ai_siege_recruit_gate_trigger = {
	OR = {
		AND = {	# покупка 1-го осадного: нужно >= 3 обычных
			NOT = { number_maa_regiments_of_base_type = { type = siege_weapon value >= 1 } }
			mormal_normal_maa_regiments_at_least_trigger = { COUNT = 3 }
		}
		AND = {	# покупка 2-го осадного: нужно >= 8 обычных
			NOT = { number_maa_regiments_of_base_type = { type = siege_weapon value >= 2 } }
			mormal_normal_maa_regiments_at_least_trigger = { COUNT = 8 }
		}
		AND = {	# покупка 3-го осадного: нужно >= 13 обычных
			NOT = { number_maa_regiments_of_base_type = { type = siege_weapon value >= 3 } }
			mormal_normal_maa_regiments_at_least_trigger = { COUNT = 13 }
		}
		AND = {	# покупка 4-го осадного: нужно >= 18 обычных
			NOT = { number_maa_regiments_of_base_type = { type = siege_weapon value >= 4 } }
			mormal_normal_maa_regiments_at_least_trigger = { COUNT = 18 }
		}
	}
}
