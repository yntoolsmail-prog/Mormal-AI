# =============================================================================
#  УСТАНОВЛЕНО ПЛЕЙТЕСТАМИ (2026-06, важнее любых догадок по файлам):
#   1. Ratio-счётчик (1 осадный на 5 обычных, vanilla/00_ai.txt:639-653) задаёт
#      только МИНИМУМ/приоритет покупки, сверху количество НЕ ограничено —
#      покупка идёт по общей оценке (score).
#   2. Средний постоянный вес цикл «распустил↔докупил» не лечит; высокий без
#      капа — спам онагров. Порог «бьёт топ у всех» ≈ +5000 (подтверждено).
#   3. Роспуск-обновление (худший полк → лучший) — штатная механика, не трогаем.
#   4. v7 (второй полк с империи) забракован плейтестом: слишком рано слишком
#      много осадных.
#
#  РЕШЕНИЕ v8: количество осадных полков ведёт ЭПОХА ОСАДНОЙ ТЕХНИКИ.
#   - До требушета: топ-вес (+5000) только у ЛУЧШЕГО открытого яруса
#     (онагр, пока нет мангонели; затем мангонель) → 1 осадный полк.
#   - С инновации требушета: топ-вес у ДВУХ лучших открытых ярусов
#     (требушет+мангонель; с бомбардой — бомбарда+требушет) → 2 полка.
#   - Устаревшие ярусы → −1000 для ИИ (v8.1, «защита от дурака»): первая
#     жертва ЛЮБОГО роспуска (бюджетный сброс/упор в лимит), докупка
#     невозможна. Замечание плейтеста: мелкие герцоги (особенно строители)
#     живут с недобранным лимитом, состояние «купить нельзя» у них редкое —
#     полк-сирота старого яруса мог висеть годами; теперь он уходит при
#     первой же встряске. Принудительный роспуск из пульса — ждёт
#     effects.log от владельца (есть ли эффект роспуска MAA вообще).
#   - max_regiments = 1 на каждый тип (ваниль-прецедент: kheshig) — спам
#     одного типа невозможен в принципе.
#   - КОЧЕВНИКИ/ГЕРДЕРЫ исключены из бустов явным условием (их ванильные
#     осадные — torch_bearers с damage=20 — не тронуты, у орды проблемы
#     не было; ванильный can_recruit этих 4 типов орде и так запрещён,
#     условие в ai_quality — страховка от экзотических путей доступа).
#
#  ai_quality читается только ИИ — баланс боя не тронут. max_regiments —
#  правило глобальное: у игрока тоже максимум 1 полк каждого осадного типа.
#
#  ПРОВЕРКА И ОТКАТ (v8):
#   - УСПЕХ: цикл мёртв; до требушета у ИИ ровно 1 осадный полк лучшего
#     яруса, после требушета — 2; замена боевых полков работает; орда
#     ведёт себя как в ванили.
#   - Переходные моменты (мангонель вытесняет онагр; бомбарда вытесняет
#     мангонель): старый полк должен распуститься ЦЕЛИКОМ и замениться;
#     если начнётся почанковый цикл на устаревшем типе — сообщить, есть
#     лекарство (запрет найма устаревших через can_recruit + is_ai).
#   - РИСК Б прежний: частые роспуски боевых («все на 20+ хуже лучшего»);
#     если так — снизить 5000 до ~1500–2000.
# =============================================================================

onager = {
	type = siege_weapon
	fights_in_main_phase = no

	damage = 0
	toughness = 12

	siege_tier = 1
	siege_value = 0.2

	can_recruit = {
		culture = {
			has_innovation = innovation_catapult
			NOR = {
				has_cultural_pillar = heritage_chinese
				has_cultural_pillar = heritage_korean
				has_cultural_pillar = heritage_japonic
			}
		}
		NOR = {
			government_has_flag = government_is_nomadic
			government_has_flag = government_is_herder
		}
	}

	should_show_when_unavailable = {
		government_allows = subject_men_at_arms
	}

	buy_cost = { gold = onager_recruitment_cost }
	low_maintenance_cost = { gold = onager_low_maint_cost }
	high_maintenance_cost = { gold = onager_high_maint_cost }
	provision_cost = @provisions_cost_special_cheap

	# MORMAL v8.1: топ-вес, пока онагр — лучший открытый ярус; устарел → −1000
	# (первая жертва любого роспуска, докупка невозможна); кочевники исключены (шапка файла)
	ai_quality = {
		value = 0
		if = {
			limit = {
				NOR = {
					government_has_flag = government_is_nomadic
					government_has_flag = government_is_herder
				}
				culture = { NOT = { has_innovation = innovation_mangonel } }
			}
			add = 5000
		}
		else_if = {
			limit = {
				NOR = {
					government_has_flag = government_is_nomadic
					government_has_flag = government_is_herder
				}
			}
			add = -1000
		}
	}

	# MORMAL: жёсткий кап — 1 полк этого типа (ваниль-прецедент: kheshig).
	# Правило глобальное: у игрока тоже максимум 1 полк каждого осадного типа.
	max_regiments = 1

	stack = 10
	allowed_in_hired_troops = no
}
