# =============================================================================
#  УСТАНОВЛЕНО ПЛЕЙТЕСТАМИ (2026-06, важнее любых догадок по файлам):
#   1. Ratio-счётчик (1 осадный на 5 обычных, vanilla/00_ai.txt:639-653) задаёт
#      только МИНИМУМ/приоритет покупки, сверху количество НЕ ограничено —
#      покупка идёт по общей оценке (score).
#   2. Средний постоянный вес цикл «распустил↔докупил» не лечит; высокий без
#      капа — спам онагров. Порог «бьёт топ у всех» ≈ +5000 (подтверждено).
#   3. Роспуск-обновление (худший полк → лучший) — штатная механика, не трогаем.
#   4. Прецедент Paradox: кочевой осадный torch_bearers (09_mpo) имеет
#      damage=20 (не 0!) — не «вечно худший», и цикла у орды не наблюдается.
#
#  РЕШЕНИЕ v6 (владелец): никакого подсчёта полков/чанков.
#   - ai_quality = 5000 ПОСТОЯННО: осадный полк — самое ценное, что ИИ может
#     завести; он заводит его сразу и доливает до полного (мёртвых слотов
#     и цикла роспуска нет — осадный никогда не «самый слабый»).
#   - max_regiments = 1 на каждый тип: больше одного полка типа нельзя
#     (ваниль-прецедент: kheshig). Хочешь больше осадных полков — открывай
#     инновации следующих ярусов (мангонель/требушет/бомбарда), каждый тоже
#     с капом 1. Это похоже на игрока: лимит полков не утонет в осаде.
#   - Лесенка весов сохранена базами: 5320 < 5420 < 5520 < 5720 — новый ярус
#     всегда ценнее старого.
#
#  ai_quality читается только ИИ — баланс боя не тронут. max_regiments —
#  правило глобальное: у игрока тоже максимум 1 полк каждого осадного типа
#  (суммарно до 4 осадных полков разных типов).
#
#  ПРОВЕРКА И ОТКАТ (v6):
#   - УСПЕХ: цикл мёртв; у ИИ ровно 1 полный полк осадных на тип; замена
#     худших боевых полков при полном лимите работает.
#   - РИСК А: с инновациями ИИ копит до 2–4 осадных полков РАЗНЫХ типов
#     (онагр+мангонель+...), т.к. старый ярус сам не распустится — его вес
#     тоже топ. Смотреть долю осадных полков в late game. Лекарство, если
#     перебор: топ-вес давать только ЛУЧШЕМУ открытому ярусу (условие по
#     has_innovation следующего яруса, подсчёт полков не нужен) — старый ярус
#     откатывается к ванили и штатно заменяется.
#   - РИСК Б: все боевые «на 20+ хуже лучшего» (осадного) → возможны частые
#     роспуски боевых. Если так — снизить 5000 до ~1500–2000.
# =============================================================================

onager = {
	type = siege_weapon
	fights_in_main_phase = no

	damage = 0
	toughness = 12

	siege_tier = 1
	siege_value = 0.2

	can_recruit = {
		culture = {
			has_innovation = innovation_catapult
			NOR = {
				has_cultural_pillar = heritage_chinese
				has_cultural_pillar = heritage_korean
				has_cultural_pillar = heritage_japonic
			}
		}
		NOR = {
			government_has_flag = government_is_nomadic
			government_has_flag = government_is_herder
		}
	}

	should_show_when_unavailable = {
		government_allows = subject_men_at_arms
	}

	buy_cost = { gold = onager_recruitment_cost }
	low_maintenance_cost = { gold = onager_low_maint_cost }
	high_maintenance_cost = { gold = onager_high_maint_cost }
	provision_cost = @provisions_cost_special_cheap

	# MORMAL v6: постоянный топ-вес — ИИ всегда хочет осадный полк (шапка файла)
	ai_quality = { value = 5000 }

	# MORMAL: жёсткий кап — 1 полк этого типа (ваниль-прецедент: kheshig).
	# Правило глобальное: у игрока тоже максимум 1 полк каждого осадного типа.
	max_regiments = 1

	stack = 10
	allowed_in_hired_troops = no
}
