#  v10.3 (решение владельца, связка Б = гейт + щит): во все 4 типа возвращён
#  ai_quality = 5000. Роль веса доказана плейтестами v9.1/v10.2: наш гейт,
#  закрываясь, обнуляет «желаемое» движка и тем самым отключает родной щит
#  от роспуска → движок ел законный полк и ИИ мерцал. Вес = замена щита;
#  количество (ровно желаемое по формуле 3/5) держит гейт, спам невозможен.
#  v10: гейт-триггер переведён на счёт ЧАНКОВ строго по движковой формуле
#  (1-й осадный чанк после 3 обычных, далее по 1 на каждые +5) и закрывается
#  только СТРОГО ВЫШЕ желаемого движком — родной донабор до желаемого и щит
#  от роспуска не затронуты. Реализация: zz_mormal_siege_chunk_values.txt
#  (чанки) + zz_mormal_siege_triggers.txt (гейт). Этот файл не менялся —
#  вставка в can_recruit та же, поменялось содержимое триггера.
#  v9.3: ai_quality=5000 из v9.2 ОТКАЧЕН по приказу владельца (вес вносился без
#  согласия; у игры есть родной щит — ИИ не распускает осадные ниже желаемого
#  числа). Диагноз цикла v9.1 пересмотрен: цикл порождает САМ ГЕЙТ — см. task.md.
#  Из v9.2 оставлены только @-константы provisions (починка Malformed token
#  из error.log владельца — восстановление ванильного поведения, не изменение).
#  v9.1: вставка в can_recruit переведена с trigger_if на OR { is_ai = no … },
#  в лесенке триггеров amount >= N заменён на amount = N — лечение симптома
#  «ИИ вообще не нанимает осадные» (подозрение на нераспознанный синтаксис).
#
#  ИСТОРИЯ (детали — task.md разд. 5): v4–v8 управляли числом осадных весами
#  (ai_quality +5000) и капом max_regiments = 1. Плейтесты показали:
#   - веса дают цикл «распустил↔докупил» или спам — управлять числом полков
#     весами нельзя;
#   - max_regiments ИИ соблюдает ненадёжно (~80%; со временем появляются
#     нарушители, нанимающие прямо сверх капа) — для ИИ-капов поле непригодно.
#  v9 (схема владельца): ВСЁ это удалено, четыре типа возвращены к ванили
#  байт-в-байт (vanilla/men_at_arms_types/00_maa_types.txt:312,422,522,564).
#  ЕДИНСТВЕННОЕ изменение — в can_recruit добавлен блок «MORMAL v9»: ИИ не
#  может нанять осадный полк сверх движковой формулы спроса (vanilla/00_ai.txt:
#  639-653): 1-й осадный — при >= 3 обычных полков, далее каждый следующий
#  при +5 обычных (2-й с 8, 3-й с 13, 4-й с 18). Ваниль покупает по этой
#  формуле МИНИМУМ, но сверху не ограничивает — гейт превращает её в потолок.
#  Проверка движковая, при каждой покупке — то, чем max_regiments не был.
#  Сами триггеры — common/scripted_triggers/zz_mormal_siege_triggers.txt.
#
#  Игрок НЕ затронут (is_ai = yes); баланс боя не тронут; кочевникам/гердерам
#  эти 4 типа запрещает сам ванильный can_recruit (их torch_bearers не трогаем);
#  Китай/Корея/Япония отсечены ванильным же условием наследия (их клоны-осадные
#  из 10_tgp НЕ перекрыты — но гейт считает осадные по базовому типу
#  siege_weapon, при необходимости тем же блоком накрываются и они).
#
#  ПРОВЕРКА И ОТКАТ (v9):
#   - УСПЕХ: ни у одного ИИ осадных полков больше формулы (герцог с 4 обычными
#     — 1 осадный; король с 8+ — до 2); «нарушители» исчезли; цикл мёртв
#     (распущенный сверхлимитный осадный не перенанимается — гейт не пускает).
#   - РИСК: ИИ с < 3 обычными полками вообще без осад (по формуле так и
#     задумано движком); если осад в мире станет слишком мало — ослабить
#     пороги в zz_mormal_siege_triggers.txt (3/8/13/18 → 2/7/12/17 и т.п.).
#   - Откат: удалить этот файл и zz_mormal_siege_triggers.txt = чистая ваниль.
# =============================================================================

	type = siege_weapon
	fights_in_main_phase = no

	damage = 0
	toughness = 12

	siege_tier = 1
	siege_value = 0.2

	can_recruit = {
		culture = {
			has_innovation = innovation_catapult
			NOR = {
				has_cultural_pillar = heritage_chinese
				has_cultural_pillar = heritage_korean
				has_cultural_pillar = heritage_japonic
			}
		}
		NOR = {
			government_has_flag = government_is_nomadic
			government_has_flag = government_is_herder
		}
		# MORMAL v9.1: ИИ не нанимает осадные сверх формулы спроса (шапка файла)
		OR = {
			is_ai = no
			mormal_ai_siege_recruit_gate_trigger = yes
		}
	}

	should_show_when_unavailable = {
		government_allows = subject_men_at_arms
	}

	buy_cost = { gold = onager_recruitment_cost }
	low_maintenance_cost = { gold = onager_low_maint_cost }
	high_maintenance_cost = { gold = onager_high_maint_cost }
	provision_cost = @provisions_cost_special_cheap

	# MORMAL v10.3 (решение владельца, связка Б): ЩИТ ОТ РОСПУСКА.
	# Доказано плейтестами v9.1/v10.2: закрытый can_recruit обнуляет «желаемое»

	# мерцание. Вес заменяет отключённый щит. Купить сверх формулы он не даёт

	allowed_in_hired_troops = no
