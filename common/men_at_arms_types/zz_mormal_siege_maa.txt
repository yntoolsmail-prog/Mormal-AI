# =============================================================================
#  Mormal AI — осадные латники: условный вес ВВЕРХ при существующем полку
#  (v5, схема владельца после плейтестов, task.md разд. 5)
# =============================================================================
#  УСТАНОВЛЕНО ПЛЕЙТЕСТАМИ (2026-06, важнее любых догадок по файлам):
#   1. Ratio-счётчик (1 осадный на 5 обычных, vanilla/00_ai.txt:639-653) задаёт
#      только МИНИМУМ/приоритет покупки. Сверху количество НЕ ограничено:
#      покупка идёт по общей оценке (score), при ai_quality +5000 ИИ собирает
#      армию преимущественно из онагров уже со старта.
#   2. Постоянный средний вес (+100…+220) цикл «распустил↔докупил» НЕ лечит:
#      движок масштабирует оценку на размер пачки/цену (пачка осадных — 10
#      человек против 100 у боевых), на скрытой шкале добавка усыхает ~в 10
#      раз. Окно «выше лёгких, ниже топов» зажато с двух сторон тестами
#      владельца (+сотни → всё ещё распускают; +5000 → спам онагров) и с
#      учётом случайных ±20% на правителя практически отсутствует.
#   3. Роспуск-обновление (худший полк → лучший) — штатная механика, её не
#      трогаем; осадный полк нельзя распустить целиком (минимум держит
#      счётчик), поэтому жертвой выбора «самый слабый» вечно выходит осадный.
#
#  v4 (минус при существующем полку) ОТКЛОНЁН владельцем без плейтеста:
#  роспуск-замена запускается только из состояния «купить никого нельзя»;
#  вечно отказываясь заполнять свободные осадные слоты, ИИ мог никогда не
#  попасть в это состояние → замена худших БОЕВЫХ полков замерла бы. Верно.
#
#  РЕШЕНИЕ v5 (схема владельца): вес осадных УСЛОВНЫЙ, но ВВЕРХ.
#   - Осадного полка ещё нет → добавка 0, всё ванильно: стартовые армии и
#     первые покупки по счётчику не затронуты (спама онагров со старта,
#     как при постоянном +5000, быть не должно — на момент генерации
#     осадного полка ещё нет).
#   - Осадный полк уже есть → добавка +5000: осадный становится САМЫМ
#     ценным (порог подтверждён тестом владельца: +5000 бьёт топ у всех).
#     Ожидание: осадный полк заполняется до конца штатной покупкой (мёртвые
#     слоты исчезают), «всё забито» достигается, роспуск-замена работает,
#     но жертвой выбора «самый слабый» осадный больше не бывает никогда.
#
#  ai_quality читается только ИИ — игрок и баланс боя не тронуты.
#
#  ПРОВЕРКА И ОТКАТ (v5):
#   - error.log при загрузке: если триггер number_maa_regiments_of_base_type
#     не существует под этим именем — лог скажет сразу (запасное имя:
#     number_maa_regiments_of_type по каждому типу отдельно).
#   - УСПЕХ: цикл мёртв; осадный полк полон; замена худших боевых полков
#     на лучшие при полном лимите продолжает работать.
#   - РИСК А: «лучший доступный» теперь осадный → при роспуске худшего
#     боевого полка освободившееся место полка может уйти под ВТОРОЙ
#     осадный полк → армия со временем дрейфует в осадные. Смотреть:
#     число осадных полков > 1, рост доли осадных в армиях ИИ.
#   - РИСК Б: все боевые теперь «на 20+ хуже лучшего» (осадного) → замена
#     боевых полков может стать слишком агрессивной (постоянные роспуски).
#   - Если Риск А/Б подтвердится → снизить добавку (например до уровня
#     «выше топ-конницы, но без запаса»: ~1500–2000) или откатить.
#
#  Тип перекрывается ЦЕЛИКОМ — ниже полные копии onager/mangonel/trebuchet/
#  bombard из vanilla/men_at_arms_types/00_maa_types.txt (CK3 1.19);
#  отличие от ванили — только блок ai_quality (помечен # MORMAL).
# =============================================================================

onager = {
	type = siege_weapon
	fights_in_main_phase = no

	damage = 0
	toughness = 12

	siege_tier = 1
	siege_value = 0.2

	can_recruit = {
		culture = {
			has_innovation = innovation_catapult
			NOR = {
				has_cultural_pillar = heritage_chinese
				has_cultural_pillar = heritage_korean
				has_cultural_pillar = heritage_japonic
			}
		}
		NOR = {
			government_has_flag = government_is_nomadic
			government_has_flag = government_is_herder
		}
	}

	should_show_when_unavailable = {
		government_allows = subject_men_at_arms
	}

	buy_cost = { gold = onager_recruitment_cost }
	low_maintenance_cost = { gold = onager_low_maint_cost }
	high_maintenance_cost = { gold = onager_high_maint_cost }
	provision_cost = @provisions_cost_special_cheap

	# MORMAL: при существующем осадном полку вес поднимается на самый верх (шапка файла)
	ai_quality = {
		value = 0
		if = {
			limit = {
				number_maa_regiments_of_base_type = { type = siege_weapon value >= 1 }
			}
			add = 5000
		}
	}

	# MORMAL v5.1: жёсткий запрет ВТОРОГО полка этого типа (лечит стартовый спам
	# онагров под бустом; ваниль-прецедент: kheshig). Глобальное правило — у
	# игрока тоже максимум 1 полк каждого осадного типа.
	max_regiments = 1

	stack = 10
	allowed_in_hired_troops = no
}
