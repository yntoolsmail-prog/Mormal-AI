# =============================================================================
#  Mormal AI — осадные латники: число осадных полков от размера армии
#  (v7, схема владельца «каждый ~5-й полк — осадный», task.md разд. 5)
# =============================================================================
#  УСТАНОВЛЕНО ПЛЕЙТЕСТАМИ (2026-06, важнее любых догадок по файлам):
#   1. Ratio-счётчик (1 осадный на 5 обычных, vanilla/00_ai.txt:639-653) задаёт
#      только МИНИМУМ/приоритет покупки, сверху количество НЕ ограничено —
#      покупка идёт по общей оценке (score).
#   2. Средний постоянный вес цикл «распустил↔докупил» не лечит; высокий без
#      капа — спам онагров. Порог «бьёт топ у всех» ≈ +5000 (подтверждено).
#   3. Роспуск-обновление (худший полк → лучший) — штатная механика, не трогаем.
#   4. Прецедент Paradox: кочевой осадный torch_bearers (09_mpo) имеет
#      damage=20 (не 0!) — не «вечно худший», и цикла у орды не наблюдается.
#
#  РЕШЕНИЕ v7: «1 осадный полк на ~5 доступных».
#  Реальный лимит полков скрипту не виден (ярус + модификаторы; модификаторы
#  не читаются), прокси — ярус титула: королевство и ниже (≈4–5 полков) →
#  1 осадный полк; империя/гегемония (≈5–10) → 2. Число осадных полков при
#  капе max_regiments = 1 равно числу типов с топ-весом, поэтому:
#   - ЛУЧШИЙ открытый ярус осадных → ai_quality +5000 всегда;
#   - ВТОРОЙ по новизне → +5000 только при highest_held_title_tier >= империя;
#   - устаревшие ярусы → 0 (ваниль ~320–520) → «худший» → полк штатно
#     распускается и заменяется боевым (это же закрывает риск v6 — накопление
#     онагр+мангонель+требушет+бомбарда у одного ИИ).
#   - max_regiments = 1 на каждый тип (ваниль-прецедент: kheshig) — спам
#     одного типа невозможен в принципе.
#
#  ai_quality читается только ИИ — баланс боя не тронут. max_regiments —
#  правило глобальное: у игрока тоже максимум 1 полк каждого осадного типа.
#
#  ПРОВЕРКА И ОТКАТ (v7):
#   - УСПЕХ: цикл мёртв; король держит 1 осадный полк лучшего яруса,
#     император — 2 (лучший + предыдущий); замена боевых полков работает.
#   - Переходный момент (открыли новый ярус → старый потерял топ-вес):
#     старый полк должен распуститься ЦЕЛИКОМ и замениться; если вместо
#     этого начнётся почанковый цикл «распустил-докупил» на устаревшем
#     типе — сообщить, есть лекарство (запрет найма устаревших через
#     can_recruit + is_ai).
#   - РИСК Б прежний: частые роспуски боевых («все на 20+ хуже лучшего»);
#     если так — снизить 5000 до ~1500–2000.
# =============================================================================

onager = {
	type = siege_weapon
	fights_in_main_phase = no

	damage = 0
	toughness = 12

	siege_tier = 1
	siege_value = 0.2

	can_recruit = {
		culture = {
			has_innovation = innovation_catapult
			NOR = {
				has_cultural_pillar = heritage_chinese
				has_cultural_pillar = heritage_korean
				has_cultural_pillar = heritage_japonic
			}
		}
		NOR = {
			government_has_flag = government_is_nomadic
			government_has_flag = government_is_herder
		}
	}

	should_show_when_unavailable = {
		government_allows = subject_men_at_arms
	}

	buy_cost = { gold = onager_recruitment_cost }
	low_maintenance_cost = { gold = onager_low_maint_cost }
	high_maintenance_cost = { gold = onager_high_maint_cost }
	provision_cost = @provisions_cost_special_cheap

	# MORMAL v7: топ-вес лучшему открытому ярусу; второму — только при империи+ (шапка файла)
	ai_quality = {
		value = 0
		if = {
			limit = {
				culture = { NOT = { has_innovation = innovation_mangonel } }
			}
			add = 5000
		}
		else_if = {
			limit = {
				culture = { NOT = { has_innovation = innovation_trebuchet } }
				highest_held_title_tier >= tier_empire
			}
			add = 5000
		}
	}

	# MORMAL: жёсткий кап — 1 полк этого типа (ваниль-прецедент: kheshig).
	# Правило глобальное: у игрока тоже максимум 1 полк каждого осадного типа.
	max_regiments = 1

	stack = 10
	allowed_in_hired_troops = no
}
